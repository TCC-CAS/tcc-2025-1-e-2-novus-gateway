    ```typescript
    import type { FastifyPluginAsync } from "fastify"
    import fastifySocketIO from "fastify-socket.io"
    import { eq, or } from "drizzle-orm"
    import { conversations } from "../db/schema/conversations.js"

    // Import the Better Auth instance — use the same import path as src/plugins/auth.ts uses
    // Check src/plugins/auth.ts for the exact auth instance import (likely: import { auth } from "../lib/auth.js")
    import { auth } from "../lib/auth.js"

    const socketIOPlugin: FastifyPluginAsync = async (fastify) => {
      // Register fastify-socket.io BEFORE route plugins
      await fastify.register(fastifySocketIO, {
        cors: {
          origin: process.env.CORS_ORIGIN || "*",
          credentials: true,
        },
      })

      // io.use() — authentication middleware (per D-07)
      // Runs for EVERY connection before "connection" event fires
      // Reject unauthenticated connections here (satisfies MSG-03b, SC-5)
      fastify.io.use(async (socket, next) => {
        try {
          const cookieHeader = socket.handshake.headers.cookie
          if (!cookieHeader) {
            return next(new Error("No session cookie"))
          }

          // Validate session using Better Auth
          const session = await auth.api.getSession({
            headers: new Headers({ cookie: cookieHeader }),
          })

          if (!session?.user) {
            return next(new Error("Invalid session"))
          }

          // Attach user to socket for use in event handlers
          socket.data.userId = session.user.id
          socket.data.role = session.user.role
          next()
        } catch {
          next(new Error("Authentication failed"))
        }
      })

      // connection handler — runs after io.use() approves the connection
      fastify.io.on("connection", async (socket) => {
        const userId = socket.data.userId as string

        // Join user's personal presence room (per D-04)
        await socket.join(`user:${userId}`)

        // Join all conversation rooms this user belongs to (per D-04, anti-pitfall-5)
        // Server-side room join on connect ensures no missed events
        const userConversations = await fastify.db
          .select({ id: conversations.id })
          .from(conversations)
          .where(
            or(
              eq(conversations.participantA, userId),
              eq(conversations.participantB, userId)
            )
          )

        for (const { id } of userConversations) {
          await socket.join(id)
        }

        // Emit user_online to all participants in user's conversation rooms (per D-03, MSG-04c)
        for (const { id } of userConversations) {
          socket.to(id).emit("user_online", { userId })
        }

        // typing_start relay (per D-03, MSG-04a)
        socket.on("typing_start", ({ conversationId }: { conversationId: string }) => {
          socket.to(conversationId).emit("typing_start", { conversationId, userId })
        })

        // typing_stop relay (per D-03, MSG-04b)
        socket.on("typing_stop", ({ conversationId }: { conversationId: string }) => {
          socket.to(conversationId).emit("typing_stop", { conversationId, userId })
        })

        // disconnect — emit user_offline to all participants (per D-03, MSG-04d)
        socket.on("disconnect", () => {
          for (const { id } of userConversations) {
            socket.to(id).emit("user_offline", { userId })
          }
        })
      })
    }

    export default socketIOPlugin
    ```

    Key notes:
    - `socket.to(room)` sends to all OTHER sockets in the room (excludes sender) — correct for typing events
    - `fastify.io.to(room)` sends to ALL sockets including sender — used by HTTP route for new_message
    - The `fastify.db` decorator is already available because db plugin is registered before socket-io in app.ts
    - If `auth.api.getSession` is not the correct call shape, check src/lib/auth.ts or src/plugins/auth.ts
      for how getSession is actually called in the existing codebase (may be `auth.getSession` or different)
