**GET /moderation/reports — list reports with pagination (ADM-03):**
```typescript
fastify.withTypeProvider<ZodTypeProvider>().get(
  "/moderation/reports",
  {
    preHandler: [requireRole("admin")],
    schema: { querystring: ListReportsQuerySchema },
  },
  async (request, reply) => {
    const { page, pageSize, status } = request.query
    const offset = Math.max(0, (page - 1) * pageSize)

    const whereClause = status ? eq(moderationReports.status, status) : undefined

    const [results, [{ value: total }]] = await Promise.all([
      fastify.db
        .select({
          id: moderationReports.id,
          reporterId: moderationReports.reporterId,
          reporterName: users.name,
          reportedEntityType: moderationReports.reportedEntityType,
          reportedEntityId: moderationReports.reportedEntityId,
          reason: moderationReports.reason,
          description: moderationReports.description,
          status: moderationReports.status,
          createdAt: moderationReports.createdAt,
        })
        .from(moderationReports)
        .leftJoin(users, eq(moderationReports.reporterId, users.id))
        .where(whereClause)
        .offset(offset)
        .limit(pageSize),
      fastify.db.select({ value: count() }).from(moderationReports).where(whereClause),
    ])

    return reply.send({
      data: results.map((r) => ({
        ...r,
        reporterName: r.reporterName ?? "Unknown",
        createdAt: r.createdAt.toISOString(),
      })),
      meta: {
        page,
        pageSize,
        total: Number(total),
        totalPages: Math.ceil(Number(total) / pageSize),
      },
    })
  }
)
```

**POST /moderation/reports/:id — take moderation action (ADM-04, per D-05/D-06/D-07/D-08):**
```typescript
fastify.withTypeProvider<ZodTypeProvider>().post(
  "/moderation/reports/:id",
  {
    preHandler: [requireRole("admin")],
    schema: {
      params: z.object({ id: z.string() }),
      body: ModerateReportRequestSchema,
    },
  },
  async (request, reply) => {
    const { id } = request.params
    const { action, note } = request.body
    const adminId = request.session!.user.id

    // Fetch report
    const [report] = await fastify.db
      .select()
      .from(moderationReports)
      .where(eq(moderationReports.id, id))
    if (!report) throw new AppError(404, "NOT_FOUND", "Report not found")

    const newStatus = action === "dismiss" ? "dismissed" : "resolved"

    if (action === "dismiss") {
      // D-05: Set status to dismissed, entity untouched
      await fastify.db
        .update(moderationReports)
        .set({ status: "dismissed" })
        .where(eq(moderationReports.id, id))

      await fastify.db.insert(auditLogs).values({
        adminId,
        action: "dismiss_report",
        targetEntityType: "report",
        targetEntityId: id,
        note: note ?? null,
      })
    } else if (action === "remove") {
      // D-06: Soft-delete reported entity, set report to resolved
      await fastify.db
        .update(moderationReports)
        .set({ status: "resolved" })
        .where(eq(moderationReports.id, id))

      // Soft-delete the reported entity based on type
      // IMPORTANT: Check actual schema field names before using — read schema files in read_first
      if (report.reportedEntityType === "player") {
        await fastify.db
          .update(players)
          .set({ hidden: true } as any)
          .where(eq(players.id, report.reportedEntityId))
      } else if (report.reportedEntityType === "team") {
        await fastify.db
          .update(teams)
          .set({ hidden: true } as any)
          .where(eq(teams.id, report.reportedEntityId))
      } else if (report.reportedEntityType === "message") {
        await fastify.db
          .update(messages)
          .set({ deleted: true } as any)
          .where(eq(messages.id, report.reportedEntityId))
      }
      // If the schema field doesn't exist yet (hidden/deleted boolean), add it to the schema
      // and run `npx drizzle-kit push` before testing.

      await fastify.db.insert(auditLogs).values({
        adminId,
        action: "remove_report",
        targetEntityType: report.reportedEntityType,
        targetEntityId: report.reportedEntityId,
        note: note ?? null,
      })
    } else if (action === "warn") {
      // D-07: Increment warnCount on the user tied to the reported entity,
      // set report status to resolved, log audit (D-08)
      await fastify.db
        .update(moderationReports)
        .set({ status: "resolved" })
        .where(eq(moderationReports.id, id))

      // The reported entity's owner userId — we need to find the user from the entity.
      // For "player" and "team" entities, look up their userId from the profile table.
      // For "message" entities, look up the senderId from the messages table.
      // This requires reading the players/teams/messages schema for the userId field.
      // Implement a helper function getEntityUserId(fastify, entityType, entityId): Promise<string | null>
      // that queries the correct table and returns the associated userId.

      // Simplified approach: query directly
      let targetUserId: string | null = null
      if (report.reportedEntityType === "player") {
        const { players: playersTable } = await import("../db/schema/players.js")
        const [p] = await fastify.db
          .select({ userId: (playersTable as any).userId })
          .from(playersTable)
          .where(eq(playersTable.id, report.reportedEntityId))
        targetUserId = p?.userId ?? null
      } else if (report.reportedEntityType === "team") {
        const { teams: teamsTable } = await import("../db/schema/teams.js")
        const [t] = await fastify.db
          .select({ userId: (teamsTable as any).userId })
          .from(teamsTable)
          .where(eq(teamsTable.id, report.reportedEntityId))
        targetUserId = t?.userId ?? null
      } else if (report.reportedEntityType === "message") {
        const { messages: messagesTable } = await import("../db/schema/messages.js")
        const [m] = await fastify.db
          .select({ senderId: (messagesTable as any).senderId })
          .from(messagesTable)
          .where(eq(messagesTable.id, report.reportedEntityId))
        targetUserId = m?.senderId ?? null
      }

      if (targetUserId) {
        // Increment warnCount using Drizzle sql template (per D-07)
        const { sql } = await import("drizzle-orm")
        await fastify.db
          .update(users)
          .set({ warnCount: sql`${users.warnCount} + 1` })
          .where(eq(users.id, targetUserId))
      }

      await fastify.db.insert(auditLogs).values({
        adminId,
        action: "warn_user",
        targetEntityType: report.reportedEntityType,
        targetEntityId: report.reportedEntityId,
        note: note ?? null,
      })
    }

    return reply.send({ data: { id, status: newStatus } })
  }
)
```
